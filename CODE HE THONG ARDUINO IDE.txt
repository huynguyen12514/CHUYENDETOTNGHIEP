#define BLYNK_TEMPLATE_ID "TMPL6HbYw7dWR"
#define BLYNK_TEMPLATE_NAME "CDTNHethongtuoinuocthongminhIoT"
#define BLYNK_AUTH_TOKEN "WAqDl20qBrr-YBDqiL09njva3s_T9pHG"

#include <ESP8266WiFi.h>
#include <BlynkSimpleEsp8266.h>
#include <EEPROM.h>
#include <WiFiManager.h> 
#include "DHT.h" 
#include <TimeLib.h>           // Thư viện quản lý thời gian
#include <WidgetRTC.h>         // Thư viện lấy thời gian thực từ Internet

// --- CẤU HÌNH PHẦN CỨNG ---
const int sensorPin = A0;      
const int pumpPin = D5;        
const int flowSensorPin = D2;  
#define DHTPIN D6              
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);
WidgetRTC rtc;                 // Khởi tạo bộ lấy giờ thực

// --- BIẾN TOÀN CỤC ---
volatile long pulseCount = 0;
float flowRate = 0.0;
float totalLiters = 0.0;
unsigned long oldTime = 0;
unsigned long lastSendBlynk = 0; 
bool isWarned = false; 

// Biến quản lý thời gian tưới ưu tiên
long startSecond, stopSecond; 
bool isTimePriority = false;   // Cờ kiểm tra có đang trong giờ ưu tiên không

struct PlantProfile {
  int startT; 
};

PlantProfile library[14] = {
  {0},  
  {45}, // 1. Cam / Quyt
  {40}, // 2. Xoai
  {55}, // 3. Dua hau / Dua luoi
  {50}, // 4. Vu sua
  {65}, // 5. Rau muong
  {45}, // 6. Rau cai / Xa lach
  {50}, // 7. Ca chua / Ot
  {45}, // 8. Kho qua / Dua leo
  {30}, // 9. Xuong rong / Sen da
  {50}, // 10. Hoa hong
  {60}, // 11. Hoa lan
  {35}, // 12. Hoa giay / Luoi ho
  {40}  // 13. Cay trong tuy chinh (Mac dinh 40%)
};
int activeStart, activeStop, selectedPlantID = 1;
int inputThreshold = 0; 
int compensator = 0;    

void IRAM_ATTR pulseCounter() { pulseCount++; }

void loadData(int id) {
  if (id < 1 || id > 13) id = 1;
  selectedPlantID = id;
  EEPROM.write(50, id); 
  int s = EEPROM.read(100 + (id * 2)); 
  activeStart = (s > 0 && s <= 100) ? s : library[id].startT;
  activeStop = activeStart + 25; 
  EEPROM.commit();
}

// --- XỬ LÝ BLYNK ---

// MỚI: Nhận khung giờ ưu tiên từ Widget Time Input trên V2
BLYNK_WRITE(V2) {
  TimeInputParam t(param);
  if (t.hasStartTime()) {
    startSecond = (t.getStartHour() * 3600) + (t.getStartMinute() * 60);
    stopSecond = (t.getStopHour() * 3600) + (t.getStopMinute() * 60);
    Serial.println("Đã cập nhật khung giờ ưu tiên!");
  }
}

BLYNK_WRITE(V4) { 
  loadData(param.asInt()); 
  Blynk.virtualWrite(V3, activeStart); 
}

BLYNK_WRITE(V9) { inputThreshold = param.asInt(); } 

BLYNK_WRITE(V5) { 
  if (param.asInt() == 1 && inputThreshold > 0) {
    EEPROM.write(100 + (selectedPlantID * 2), inputThreshold);
    EEPROM.commit();
    activeStart = inputThreshold; 
    activeStop = activeStart + 25;
    Blynk.virtualWrite(V3, activeStart);
  }
}

BLYNK_WRITE(V10) { 
  if (param.asInt() == 1) {
    EEPROM.write(100 + (selectedPlantID * 2), 255); 
    EEPROM.commit();
    loadData(selectedPlantID);
    Blynk.virtualWrite(V3, activeStart);
  }
}

BLYNK_WRITE(V6) { compensator = param.asInt(); } 

BLYNK_WRITE(V7) { 
  if (param.asInt() == 1) { 
    totalLiters = 0; 
    Blynk.virtualWrite(V8, 0); 
  } 
}

void setup() {
  Serial.begin(115200);
  EEPROM.begin(512);
  dht.begin(); 
  
  pinMode(pumpPin, OUTPUT);
  digitalWrite(pumpPin, HIGH); 
  
  pinMode(flowSensorPin, INPUT_PULLUP);
  attachInterrupt(digitalPinToInterrupt(flowSensorPin), pulseCounter, FALLING);

  WiFiManager wm;
  if (!wm.autoConnect("CDTN_64130859_HTTNTMIoT", "12345678")) {
    delay(5000);
    ESP.restart();
  }

  Blynk.config(BLYNK_AUTH_TOKEN);
  rtc.begin(); // Bắt đầu lấy giờ từ Server

  int lastPlant = EEPROM.read(50);
  selectedPlantID = (lastPlant >= 1 && lastPlant <= 13) ? lastPlant : 1;
  loadData(selectedPlantID);
  
  Blynk.virtualWrite(V3, activeStart);
  Blynk.virtualWrite(V4, selectedPlantID);
}

void loop() {
  Blynk.run();

  // Kiểm tra thời gian thực để xác định có trong giờ ưu tiên không
  long nowSecond = (hour() * 3600) + (minute() * 60) + second();
  if (startSecond < stopSecond) { // Ví dụ: 6:00 đến 18:00
    isTimePriority = (nowSecond >= startSecond && nowSecond <= stopSecond);
  } else { // Ví dụ: 22:00 đến 05:00 sáng hôm sau
    isTimePriority = (nowSecond >= startSecond || nowSecond <= stopSecond);
  }

  // 1. Tính lưu lượng
  if ((millis() - oldTime) > 1000) {
    detachInterrupt(digitalPinToInterrupt(flowSensorPin));
    flowRate = ((1000.0 / (millis() - oldTime)) * pulseCount) / 7.5;
    oldTime = millis();
    totalLiters += (flowRate / 60.0);
    pulseCount = 0;
    attachInterrupt(digitalPinToInterrupt(flowSensorPin), pulseCounter, FALLING);
  }

  // 2. Gửi dữ liệu định kỳ 5s
  if (millis() - lastSendBlynk > 5000) { 
    int percent = constrain(map(analogRead(sensorPin), 1023, 350, 0, 100), 0, 100);
    float h = dht.readHumidity();
    float t = dht.readTemperature();

    Blynk.virtualWrite(V0, percent);      
    Blynk.virtualWrite(V1, flowRate);     
    Blynk.virtualWrite(V8, totalLiters);  
    
    if (!isnan(h) && !isnan(t)) {
      Blynk.virtualWrite(V11, t);    
      Blynk.virtualWrite(V12, h);    
      
      if (t > 36.0 && !isWarned) {
          Blynk.logEvent("temp_alert", String("Nhiệt độ cao: ") + t + "°C");
          isWarned = true; 
      } else if (t <= 36.0) {
          isWarned = false; 
      }
    }
    lastSendBlynk = millis();
  }

  // 3. LOGIC ĐIỀU KHIỂN BƠM (Thêm điều kiện thời gian)
  int percentNow = constrain(map(analogRead(sensorPin), 1023, 350, 0, 100), 0, 100);
  int startLimit = activeStart + compensator;
  int stopLimit = activeStop + compensator;

  // Bơm chỉ bật khi Đất khô VÀ Đang trong giờ ưu tiên
  if (percentNow < startLimit && isTimePriority) {
      digitalWrite(pumpPin, LOW);  // Bật bơm
  } 
  // Bơm tắt khi Đất đủ ẩm HOẶC Đã hết giờ ưu tiên (để an toàn)
  else if (percentNow > stopLimit || !isTimePriority) {
      digitalWrite(pumpPin, HIGH); // Tắt bơm
  }
}
